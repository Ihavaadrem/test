# 网页第一课 作业
<!DOCTYPE HTML>
<html>
    <head>
        <meta http-equiv="Content-Type" content="text/html; charset=utf-8">
        <title>制作我的第一个网页</title>
    </head>
    <body>
        <h1>h1标签</h1>
        <h2>h2标签</h2>
        
        <br/> <!--换行-->
        <b/><!--加粗-->
        <a href="#"></a><!--超链接-->
        <a name="fist"></a><!--页面内链接-->
        <img src="#" /> <!--图片链接-->
        <form action="#" method="get&post"> <!--数据传输  get为链接后面的id=****传输；post为不显示隐藏传输，通常用于密码登录传输-->
          <input type="text"> <!--文本框-->
          <input type="password"> <!--密码框-->
          <input type="submit"> <!--按钮-->
        </form>
        <tabele> <!--表格-->
         <tr> <!-- table的一行-->
          <th></th><!--表格内文字加粗-->
          <td></td><!-- tr中的一列-->
         </tr>
        </tabele>
        <div>  <!--Div 框架-->
          <ul>
            <li>1</li>
            <li>2</li>
            <li>3</li>
          </ul>
        </div>
        </body>
</html>
